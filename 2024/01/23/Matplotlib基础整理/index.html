柱状图

  • 双向柱状图
  • 调柱子宽度
  • 调颜色
  • 数值标签
import matplotlib.pyplot as plt

# 初始化一个图形和轴对象
fig, ax = plt.subplots(figsize=(10, 7))  # figsize=(宽度, 高度)，单位为英寸，指定图形的大小

# 有5个比赛集
set_numbers = [1, 2, 3, 4, 5]  # 每个集的编号
width = 0.45  # 每个柱子的宽度

# 每位选手在每个集中赢得的游戏数
p1_games = [1, 7, 6, 3, 6]  # 玩家1赢得的游戏数
p2_games = [-6, -6, -1, -6, -4]  # 玩家2赢得的游戏数，使用负数以在x轴下方显示

# 绘制柱状图
rects1 = ax.bar(set_numbers, p1_games, width, label='Alcaraz', color='#a4e66b')  # 绘制玩家1的柱状图
rects2 = ax.bar(set_numbers, p2_games, width, label='Djokovic', color='#3fb1e3')  # 绘制玩家2的柱状图

# 设置y轴的范围
ax.set_ylim(-8, 9)
# 设置x轴和y轴的标签
ax.set_xlabel('Set Number', fontsize=23)  # 设置x轴的标签及其字体大小
ax.set_ylabel('Number of Won Games', fontsize=23)  # 设置y轴的标签及其字体大小
# 设置图表的标题
ax.set_title('Number of Won Games per Set', fontsize=23, pad=15)  # 设置标题及其字体大小和上边距

# 设置x轴的刻度
ax.set_xticks(set_numbers)  # 设置x轴上的刻度位置，与比赛集编号对应
# 添加图例
ax.legend(loc='upper left', bbox_to_anchor=(0, 1), fontsize=19)  # 设置图例的位置和字体大小

# 绘制x轴上的0线
ax.axhline(0, color='black', linewidth=0.8)  # 绘制x轴（y=0）的线
# 隐藏y轴的刻度标签
ax.set_yticklabels([])  # 将y轴上的刻度标签设置为空列表，以隐藏它们
# 设置x轴刻度标签的大小
ax.tick_params(axis='x', labelsize=23)  # 设置x轴上的刻度标签的字体大小

# 在每个柱子上方或下方添加数值标签
for rect in rects1 + rects2:
    height = rect.get_height()  # 获取柱子的高度
    # 根据柱子的位置，添加相应的数值标签
    ax.text(rect.get_x() + rect.get_width() / 2., 1.01*height, '%d' % int(abs(height)),
            ha='center', va='bottom' if height > 0 else 'top', color='black', fontsize=23)

fig.tight_layout()  # 调整子图参数，使之填充整个图形区域
plt.show()  # 显示图形
